                \begin{align*}
                    \mathop{\color{blue}\operatorname{div}} E &= 4\pi \rho & \operatorname{div} B &= 0 \\

                \[
                    S = \int \sqrt{-\color{blue}g}{\color{blue}g^{ac}g^{bd}} F_{ab} F_{cd} d^4 x
